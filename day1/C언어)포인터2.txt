#include<stbio.h>
main()		//리턴할 자료형은 컴파일러에 따라 생략하는 것도 가능
{
	char *p = "KOREA";
	printf("%s", p);		//포인터는 배열의 가장 앞을 가리킴. 즉 K의 주소가 담겨있음. 이를 제어문자 s와 함께 사용한다면 문자열의 가장 앞부터 모든 부분 출력
	printf("%s", p + 3);	//p + 3은 배열의 가장 앞에 3을 더한 위치를 의미. 이를 문자열로 출력한다면 EA를 출력하게 됨
	printf(%c", *p);		//문자열 대신 문자를 출력. *p를 통해 해당 주소의 값을 찾고, 이를 문자열로 출력. p는 맨 앞을 가리키니 해당 위치의 문자 K가 출력됨
	printf("%c", *(p+3));	//K로부터 3 뒤에 위치한 문자를 출력. 즉 E가 출력됨
	printf("%c", *p + 2);	//제어문자 c에 의해 *p는 K가 된다. 이때, 문자가 산술 연산자를 만나면 자연스럽게 아스키코드를 통한 산술연산이 시행됨. 따라서 K를 아스키코드 75로 환산 후 해당 숫자에 2를 더해줌. 즉 77을 문자로 출력하는 코드. 결과적으로 M이 출력됨
}
