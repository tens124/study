#include<stbio.h>
main()
{
	int s , el = 0;
	for(int i = 6; i <=30;++)
	{
		s = 0;

		for(int j = 1; j<=i/2;j++)	//어떤 수의 약수는 그 수를 2로 나눈 값 이하에서만 존재하기에, i/2를 통해 연산 횟수를 줄여줌(약수는 자기 자신도 포함하지만, 자기 자신은 따로 다루는 쪽이 훨씬 이롭다)
		{
			if(i%j ==0)			//나머지가 0인 수, 즉 약수를 의미함
			{	
				s=s+j;		//j가 약수라면 s에 적립해줌
			}
		}

		if(s==i)				//약수를 모두 더한 값이 원래의 수 i와 동일하다면 el을 1 올려준다
		{
			el++;
		}	
	}
}

//완전수는 스스로를 제외한 약수들의 합이 자기 자신과 동일한 수를 의미함
//완전수를 찾는 알고리즘이지만, 몇 개 외워두는 편이 훨씬 낫다
