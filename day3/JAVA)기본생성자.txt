class Example 
{
    int intValue = 42;       // 명시적 초기화
    double doubleValue = 3.14; 	// 명시적 초기화
    boolean booleanValue = true; 	// 명시적 초기화
    String stringValue = "Hello, world!"; // 명시적 초기화

    Example()  	// 묵시적 기본 생성자
			//클래스를 작성할 때 아무런 생성자를 만들지 않는다 해도 객체를 생성할 때 컴파일러에 의해 '기본 생성자'가 추가되어 초기화를 수행
			//기본 생성자는 각 필드변수에 자료형에 따른 초기값을 할당해 줌. 필드 변수가 명시적인 초기화 없이도 정상적으로 작동하는 이유	  

			//다만 필드 변수를 명시적으로 초기화하는 경우, 기본 생성자는 아무런 행동도 하지 않게 됨
			//이처럼 생성자가 작동하지 않는 것처럼 보이는 경우가 있다
			//하나는 생성자가 호출되지 않는 경우 - new 연산자 없이 객체를 호출할 때. 정적(static) 메소드 등			
			//또 하나는 위의 경우처럼 생성자의 작업이 필드 초기화에 영향을 미치지 않을 때
	{
        // 아무것도 하지 않음. 필드는 이미 명시적으로 초기화되었기 때문에 기본 생성자가 영향을 미치지 않는다	
	}
}
