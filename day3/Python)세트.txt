a = {'apple', 'lemon', 'banana'}	#세트 자료형. 리스트는 [], 세트는 {}
						#세트는 순서가 없고 중복 자료형을 저장할 수 없음
a.update( {'kiwi', 'banana'} )		#update( { } ) 메소드 : 세트에 새로운 세트를 추가하여 확장. 중복값을 제하고 남은 값을 추가
a.remove('lemon')				#remove(내용) 메소드 : 세트에서 해당 내용을 제거
a.add('apple')				#add(내용) 메소드 : 세트에 값을 추가. 중복값이니 변화 X
for i in a :					#for문 사용 시 in 뒤에 Range  대신 자료구조를 사용 가능. 해당 자료구조의 모든 요소를 순회하며, 각 차례마다 해당 요소의 값을 i에 저장
	print("과일명 : %s" % i)		#파이썬에서는 형식문자를 이렇게 사용. print메소드는 동일하며, "" 안에 형식문자와 함께 출력 내용을 서술. 그 후 % 를 통해 해당 형식문자에 출력할 내용을 입력해줌
